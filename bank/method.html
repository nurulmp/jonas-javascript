JavaScript Array Methods Cheat Sheet (বাংলায়)

মেথড (Method) কাজ (Purpose) উদাহরণ (Example)
push() অ্যারের শেষে একটি আইটেম যোগ করে [1, 2].push(3) → [1, 2, 3]
pop() অ্যারের শেষ আইটেম বাদ দেয় [1, 2, 3].pop() → [1, 2]
shift() অ্যারের প্রথম আইটেম বাদ দেয় [1, 2, 3].shift() → [2, 3]
unshift() অ্যারের শুরুতে আইটেম যোগ করে [2, 3].unshift(1) → [1, 2, 3]
includes() অ্যারে নির্দিষ্ট মান আছে কিনা চেক করে [1, 2].includes(2) → true
indexOf() অ্যারে নির্দিষ্ট মানের ইনডেক্স দেয় [1, 2].indexOf(2) → 1
find() শর্ত অনুযায়ী প্রথম মিলে যাওয়া আইটেম খুঁজে আনে [1, 3].find(x => x > 2) → 3
findIndex() শর্ত অনুযায়ী প্রথম মিলে যাওয়া আইটেমের ইনডেক্স দেয় [1, 3].findIndex(x => x > 2) → 1
some() অন্তত একটি আইটেম শর্ত পূরণ করে কিনা পরীক্ষা করে [1, 2].some(x => x > 1) → true
every() সব আইটেম শর্ত পূরণ করে কিনা পরীক্ষা করে [2, 4].every(x => x % 2 === 0) → true
forEach() অ্যারের প্রতিটি আইটেমে লুপ চালায় [1, 2].forEach(x => console.log(x))
map() প্রতিটি আইটেমকে পরিবর্তন করে নতুন অ্যারে তৈরি করে [1, 2].map(x => x * 2) → [2, 4]
filter() শর্ত পূরণকারী আইটেমগুলো দিয়ে নতুন অ্যারে তৈরি করে [1, 2, 3].filter(x => x > 1) → [2, 3]
reduce() অ্যারের সব আইটেমকে একক মানে রূপান্তর করে [1, 2, 3].reduce((a, b) => a + b) → 6
slice() অ্যারের অংশবিশেষ কেটে আনে [1, 2, 3].slice(1) → [2, 3]
splice() অ্যারে থেকে আইটেম সরায় বা যোগ করে arr.splice(1, 1) → index 1 থেকে সরায়
concat() দুই বা তার বেশি অ্যারে যুক্ত করে [1, 2].concat([3]) → [1, 2, 3]
flat() নেস্টেড অ্যারে সমান স্তরে নিয়ে আসে [1, [2, 3]].flat() → [1, 2, 3]
fill() অ্যারের সব ঘর একই মান দিয়ে ভরাট করে [1, 2, 3].fill(0) → [0, 0, 0]
reverse() অ্যারের আইটেম উল্টো করে [1, 2].reverse() → [2, 1]
sort() অ্যারের আইটেম সাজায় [3, 1].sort() → [1, 3]
Array.from() ইটারেবল অবজেক্ট থেকে অ্যারে তৈরি করে Array.from('abc') → ['a', 'b', 'c']
Array.of() নির্দিষ্ট মান দিয়ে অ্যারে তৈরি করে Array.of(1, 2) → [1, 2]
join() অ্যারের আইটেম দিয়ে একটি স্ট্রিং তৈরি করে [1, 2].join('-') → '1-2'
toString() অ্যারের আইটেম স্ট্রিং এ রূপান্তর করে [1, 2].toString() → '1,2'
